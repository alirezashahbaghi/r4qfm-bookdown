2.2 Ordinal

rank <- c(4, 2, 3, 1, 6, 5)

temp <- c("High", "Low", "High", "Medium")
rank <- c(4, 2, 3, 1, 6, 5)

rank
[1] 4 2 3 1 6 5
class(rank)
[1] "numeric"
rank <- c(4, 2, 3, 1, 6, 5)

rank.f <- factor(rank)

rank.f
[1] 4 2 3 1 6 5
Levels: 1 2 3 4 5 6
class(rank.f)
[1] "factor"
rank <- c(4, 2, 3, 1, 6, 5)
rank.f <- factor(rank, 
                 order = TRUE, 
                 labels = c("One", "Two",
                            "Three", "four", 
                            "five", "six"))
rank.f
[1] four  Two   Three One   six   five 
Levels: One < Two < Three < four < five < six
class(rank.f)
[1] "ordered" "factor"
